  as select from ZCDS_A06_EX13 as _Sbook
  association [1..1] to ZCDS_A06_EX11 as _Scarr     on _Scarr.Carrid = _Sbook.Carrid
  association [1..1] to ZCDS_A06_EX12 as _ClassText on _ClassText.Class = _Sbook.Class

  key Fldate,

      _Scarr.Carrname                                                                            as Carrname,
