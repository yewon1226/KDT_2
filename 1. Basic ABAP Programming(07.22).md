CLASS zcl_a06_w03 IMPLEMENTATION.
  METHOD if_oo_adt_classrun~main.
      TYPES: BEGIN OF ts_temp,
               month(2),
               mcount TYPE i,
             END OF ts_temp.
      TYPES tt_temp TYPE TABLE OF ts_temp WITH KEY month.
      DATA lt_temp TYPE tt_temp.
  
      TYPES: BEGIN OF ts_sbook,
               month01 TYPE i,
               month02 TYPE i,
               month03 TYPE i,
               month04 TYPE i,
               month05 TYPE i,
               month06 TYPE i,
               month07 TYPE i,
               month08 TYPE i,
               month09 TYPE i,
               month10 TYPE i,
               month11 TYPE i,
               month12 TYPE i,
             END OF ts_sbook.
      TYPES tt_sbook TYPE TABLE OF ts_sbook.
      DATA lt_sbook TYPE tt_sbook.
  
      lt_temp = VALUE #(
          ( month = '01' mcount = 10 )
          ( month = '02' mcount = 0 )
          ( month = '03' mcount = 101 )
          ( month = '04' mcount = 19 )
          ( month = '05' mcount = 3 )
          ( month = '06' mcount = 7 )
          ( month = '07' mcount = 0 )
          ( month = '08' mcount = 39 )
          ( month = '09' mcount = 777 )
          ( month = '10' mcount = 34 )
          ( month = '11' mcount = 2 )
          ( month = '12' mcount = 111 )
      ).
  
      FIELD-SYMBOLS <f_month> TYPE any.
  
      LOOP AT lt_temp INTO DATA(ls_temp).
          DATA(lv_num) = ls_temp-month.
          DATA(lv_month) = 'LS_SBOOK-MONTH' && lv_num.
          ASSIGN (lv_month) TO <f_month>.
          <f_month> = ls_temp-mcount.
          CLEAR ls_temp.
          UNASSIGN <f_month>.
      ENDLOOP.
    ENDMETHOD.
ENDCLASS.
